    x = x * 0.6214
    return x
mil = int(input())

def get_days(x):
    if (x < 8 and x % 2 == 1) or (x >= 8 and x % 2 == 0):
        return 31
    elif x == 2:
        return 28
    else:
        return 30
num = int(input())
print(get_days(num))
