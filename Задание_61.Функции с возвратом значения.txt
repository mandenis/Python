    d = b ** 2 - 4 * a * c   
    import math as z
    if d > 0:
        number = (- b - z.sqrt(d)) / (2 * a)
        number2 = (- b + z.sqrt(d)) / (2 * a)
        if number < number2:
            return number, number2
        else:
            return number2, number
    elif d == 0:
        return - b / (2 * a), - b / (2 * a)
